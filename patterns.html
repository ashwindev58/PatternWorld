Basic Patterns

Single Star


#include <stdio.h>

int main() {
    printf("*\n");
    return 0;
}
*

Square of Stars


#include <stdio.h>

int main() {
    for(int i = 0; i < 5; i++) {
        for(int j = 0; j < 5; j++) {
            printf("*");
        }
        printf("\n");
    }
    return 0;
}

Intermediate Patterns

Pyramid


#include <stdio.h>

int main() {
    int rows = 5;
    for(int i = 1; i <= rows; i++) {
        for(int space = 1; space <= rows-i; space++) {
            printf(" ");
        }
        for(int j = 1; j <= 2*i-1; j++) {
            printf("*");
        }
        printf("\n");
    }
    return 0;
}

Inverted Pyramid


#include <stdio.h>

int main() {
    int rows = 5;
    for(int i = rows; i >= 1; i--) {
        for(int space = 1; space <= rows-i; space++) {
            printf(" ");
        }
        for(int j = 1; j <= 2*i-1; j++) {
            printf("*");
        }
        printf("\n");
    }
    return 0;
}

Advanced Patterns

Diamond


#include <stdio.h>

int main() {
    int n = 5;
    int space = n - 1;

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= space; j++)
            printf(" ");
        space--;
        for (int j = 1; j <= 2*i-1; j++)
            printf("*");
        printf("\n");
    }

    space = 1;
    for (int i = 1; i <= n-1; i++) {
        for (int j = 1; j <= space; j++)
            printf(" ");
        space++;
        for (int j = 1; j <= 2*(n-i)-1; j++)
            printf("*");
        printf("\n");
    }
    return 0;
}

Hollow Square


#include <stdio.h>

int main() {
    int size = 5;
    for(int i = 0; i < size; i++) {
        for(int j = 0; j < size; j++) {
            if (i == 0 || i == size-1 || j == 0 || j == size-1) {
                printf("*");
            } else {
                printf(" ");
            }
        }
        printf("\n");
    }
    return 0;
}

Z Pattern

int main() {
    int size = 5;
    for(int i = 0; i < size; i++) {
        for(int j = 0; j < size; j++) {

F Pattern


#include <stdio.h>

int main() {
    int rows = 5;
    for(int i = 0; i < rows; i++) {
        for(int j = 0; j < rows; j++) {
            if (i == 0 || i == rows/2 || (j == 0 && i < rows)) {
                printf("*");
            } else {
                printf(" ");
            }
        }
        printf("\n");
    }
    return 0;
}
